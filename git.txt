git status - узнать есть ли не сохраненные файлы
git add --all - добавить все не сохраненные файлы
git commit -a -m"add script fil" - сохранение контрольной точки и добавление новый файл с сохранением сообщения - описания о нем
git commit -a - сохранение контрольной точки БЕЗ добавления новый файл с сохранением сообщения - описания о нем
git push - добавить все несохраненные  файлы в гитхаб
